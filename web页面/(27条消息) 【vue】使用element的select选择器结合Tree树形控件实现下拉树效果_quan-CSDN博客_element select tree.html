本人使用的基本全是element 原本的语法实现的效果，代码比较精简。实现的效果如下，给需要的人一个参考

实现的代码：

  1.             <el-select
  2.               v-model="saveForm.upResId"
  3.               clearable
  4.               placeholder="请选择"
  5.               // 点击清空的实现
  6.               @clear="handleClear"
  7.               // ref属性注册，用于在vue中操作dom元素
  8.               ref="selectUpResId"
  9.             >
  10.               // 设置一个隐藏的下拉选项
  11.               // 如果不设置一个下拉选项，下面的树形组件将无法正常使用
  12.               <el-option hidden key="upResId" :value="saveForm.upResId" :label="upResName">
  13.               </el-option>
  14.               // 设置树形组件  
  15.               <el-tree
  16.                 :data="data"
  17.                 :props="defaultProps"
  18.                 // 是否在点击节点的时候展开或者收缩节点， 默认值为 true，如果为 false，则只有点箭头图标的时候才会展开或者收缩节点。
  19.                 :expand-on-click-node="false"
  20.                 // 是否在点击节点的时候选中节点，默认值为 false，即只有在点击复选框时才会选中节点。
  21.                 :check-on-click-node="true"
  22.                 // 节点被点击时的回调
  23.                 @node-click="handleNodeClick"
  24.               >
  25.               </el-tree>
  26.             </el-select>
  1. // data 部分：
  2. // 树形组件测试使用的数据
  3. data: [{
  4.         resId: 1,
  5.         name: '一级 1',
  6.         children: [{
  7.           resId: 11,
  8.           name: '二级 1-1',
  9.           children: [{
  10.             name: '三级 1-1-1'
  11.           }]
  12.         }]
  13.       }, {
  14.         resId: 2,
  15.         name: '一级 2',
  16.         children: [{
  17.           resId: 22,
  18.           name: '二级 2-1',
  19.           children: [{
  20.             resId: 221,
  21.             name: '三级 2-1-1'
  22.           }]
  23.         }, {
  24.           resId: 23,
  25.           name: '二级 2-2',
  26.           children: [{
  27.             resId: 224,
  28.             name: '三级 2-2-1'
  29.           }]
  30.         }]
  31.       }],
  32.       defaultProps: {
  33.         children: 'children',
  34.         label: 'name'
  35.       }
  36.  
  37.  
  38. // 函数部分    
  39.     // 节点点击事件
  40.     handleNodeClick(data) {
  41.       // 这里主要配置树形组件点击节点后，设置选择器的值；自己配置的数据，仅供参考
  42.       this.upResName = data.name
  43.       this.saveForm.upResId = data.resId
  44.       // 选择器执行完成后，使其失去焦点隐藏下拉框的效果
  45.       this.$refs.selectUpResId.blur()
  46.     },
  47.     // 选择器配置可以清空选项，用户点击清空按钮时触发
  48.     handleClear() {
  49.       // 将选择器的值置空
  50.       this.upResName = ''
  51.       this.saveForm.upResId = ''
  52.     }
