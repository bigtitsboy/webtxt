在使用 element-ui 的 el-cascader 组件根据后台返回的数据，需要展示一个默认值，官网给出的例子https://element.eleme.cn/2.0/#/zh-CN/component/cascader 借鉴了一下，话不多说直接上代码：

效果图：

代码块：

  1. <template>
  2.     <div class="block">
  3.         <span class="demonstration">默认 click 触发子菜单</span>
  4.         <el-cascader
  5.             :props="defaultParams"
  6.             :options="options"
  7.             v-model="selectedOptions">
  8.         </el-cascader>
  9.     </div>
  10. </template>
  11. <script>
  12. export default {
  13.     data () {
  14.         return {
  15.             options: [],
  16.             selectedOptions: ['1', '4'],
  17.             defaultParams: {
  18.                 label: 'name',
  19.                 value: 'id',
  20.                 children: 'sub'
  21.             },
  22.         }
  23.     },
  24.     create () {
  25.         // 这里是接口返回的数据
  26.         let options = {
  27.             "data": [{
  28.                 "id": "1",
  29.                 "name": "水果",
  30.                 "level": "1",
  31.                 "pid": "0",
  32.                 "status": "1",
  33.                 "sub": [{
  34.                     "id": "4",
  35.                     "name": "苹果",
  36.                     "level": "2",
  37.                     "pid": "1",
  38.                     "status": "1"
  39.                 }, {
  40.                     "id": "8",
  41.                     "name": "香蕉",
  42.                     "level": "2",
  43.                     "pid": "1",
  44.                     "status": "1"
  45.                 }]
  46.             }, {
  47.                 "id": "2",
  48.                 "name": "食品",
  49.                 "level": "1",
  50.                 "pid": "0",
  51.                 "status": "1",
  52.                 "sub": [{
  53.                     "id": "5",
  54.                     "name": "馒头",
  55.                     "level": "2",
  56.                     "pid": "2",
  57.                     "status": "1"
  58.                 }, {
  59.                     "id": "6",
  60.                     "name": "大米",
  61.                     "level": "2",
  62.                     "pid": "2",
  63.                     "status": "1"
  64.                 }]
  65.             }]
  66.         }
  67.         this.options = options.data
  68.     }
  69. }
  70. </script>

到这里基本上算是完成了。

注意：如果说后端返回的 id 是 String 类型，这里的默认值也是 String：selectedOptions: ['1', '4']，如果返回的是 Number 类型，默认值 selectedOptions: [1, 4]
